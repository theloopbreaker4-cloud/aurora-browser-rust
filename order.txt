Что здесь важно:

wry — это библиотека на Rust для создания браузерных окон с WebView.

В Attributes можно задать стартовую страницу, тему, иконки.

Дальше ты можешь расширять: вкладки, закладки, кастомный интерфейс.

Если хочешь более «чистый» движок, можно копнуть в Servo — это проект на Rust, который задумывался как полноценный браузерный движок. Но он сложнее для сборки и интеграции.

use wry::{Application, Attributes, Result};

fn main() -> Result<()> {
    // Создаём приложение
    let app = Application::new()?;

    // Настраиваем окно браузера
    let window = app.add_window(Attributes {
        title: String::from("Aurora"),
        url: Some(String::from("https://www.google.com")),
        ..Default::default()
    })?;

    // Запускаем цикл
    app.run();
    Ok(())
}

use wry::{Application, Attributes, Result};

struct Tab {
    title: String,
    url: String,
}

fn main() -> Result<()> {
    // Список вкладок
    let tabs = vec![
        Tab { title: "YouTube".into(), url: "https://youtube.com".into() },
        Tab { title: "Google".into(), url: "https://google.com".into() },
        Tab { title: "Wikipedia".into(), url: "https://wikipedia.org".into() },
        Tab { title: "GitHub".into(), url: "https://github.com".into() },
    ];

    // Запускаем приложение
    let app = Application::new()?;

    // Открываем первую вкладку как стартовую
    app.add_window(Attributes {
        title: String::from("Aurora"),
        url: Some(tabs[0].url.clone()),
        ..Default::default()
    })?;

    app.run();
    Ok(())
}

use wry::{Application, Attributes, Result};
use std::fs;
use serde::Deserialize;
use std::collections::HashMap;

#[derive(Deserialize)]
struct Bookmarks(HashMap<String, String>);

fn main() -> Result<()> {
    // Загружаем закладки из JSON
    let data = fs::read_to_string("bookmarks.json").unwrap();
    let bookmarks: Bookmarks = serde_json::from_str(&data).unwrap();

    // Берём первую закладку как стартовую
    let first_url = bookmarks.0.values().next().unwrap().clone();

    let app = Application::new()?;
    app.add_window(Attributes {
        title: String::from("Aurora"),
        url: Some(first_url),
        ..Default::default()
    })?;
    app.run();
    Ok(())
}

async function loadBookmarks() {
  const response = await fetch("bookmarks.json");
  const bookmarks = await response.json();

  const container = document.getElementById("bookmarks");
  container.innerHTML = "";

  for (const [title, url] of Object.entries(bookmarks)) {
    const link = document.createElement("a");
    link.href = url;
    link.textContent = title;
    link.className = "artifact";
    container.appendChild(link);
  }
}

Расширение script.js с горячими клавишами
Что это даёт
Каждая закладка получает номер (1, 2, 3…).

При нажатии Ctrl+1, Ctrl+2 и т.д. портал сразу открывает соответствующую закладку.

Это превращает навигацию в ритуальные команды — быстрые, символические переходы.
-----------------------------------------------------------------------------------------------------------------
async function loadBookmarks() {
  const response = await fetch("bookmarks.json");
  const bookmarks = await response.json();

  const container = document.getElementById("bookmarks");
  container.innerHTML = "";

  let index = 1;
  for (const [title, url] of Object.entries(bookmarks)) {
    const link = document.createElement("a");
    link.href = url;
    link.textContent = `${index}. ${title}`;
    link.className = "artifact";
    container.appendChild(link);
    index++;
  }

  // Горячие клавиши Ctrl+1, Ctrl+2, ...
  document.addEventListener("keydown", (event) => {
    if (event.ctrlKey) {
      const num = parseInt(event.key);
      if (!isNaN(num) && num > 0 && num <= Object.keys(bookmarks).length) {
        const targetUrl = Object.values(bookmarks)[num - 1];
        window.location.href = targetUrl;
      }
    }
  });
}

loadBookmarks();

use wry::{Application, Attributes, Result};
use rusqlite::{params, Connection};
use std::time::{SystemTime, UNIX_EPOCH};

fn log_visit(conn: &Connection, url: &str) {
    let timestamp = SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .unwrap()
        .as_secs();

    conn.execute(
        "INSERT INTO history (url, timestamp) VALUES (?1, ?2)",
        params![url, timestamp],
    ).unwrap();
}

fn main() -> Result<()> {
    // Создаём/открываем базу данных
    let conn = Connection::open("aurora_history.db").unwrap();
    conn.execute(
        "CREATE TABLE IF NOT EXISTS history (
            id INTEGER PRIMARY KEY,
            url TEXT NOT NULL,
            timestamp INTEGER NOT NULL
        )",
        [],
    ).unwrap();

    // Пример: логируем стартовую страницу
    let start_url = "https://startpage.com";
    log_visit(&conn, start_url);

    // Запускаем браузер
    let app = Application::new()?;
    app.add_window(Attributes {
        title: String::from("Aurora"),
        url: Some(start_url.into()),
        ..Default::default()
    })?;
    app.run();
    Ok(())
}
